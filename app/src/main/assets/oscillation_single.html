    view, SpringAnimation.ROTATION
  )
  .setSpring(
    SpringForce()
      .setFinalPosition(0f)
      .setDampingRatio(
        SpringForce.DAMPING_RATIO_HIGH_BOUNCY
      )
      .setStiffness(
        SpringForce.STIFFNESS_LOW
      )
  )
  .addUpdateListener { _, _, velocity ->
    currentVelocity = velocity
  }

view.pivotX = view.width / 2f
view.pivotY = 0f
